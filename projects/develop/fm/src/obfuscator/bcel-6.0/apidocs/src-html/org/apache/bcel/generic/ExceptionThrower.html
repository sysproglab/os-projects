018package org.apache.bcel.generic;
019
020/**
021 * Denote an instruction that may throw a run-time or a linking
022 * exception (or both) during execution.  This is not quite the truth
023 * as such; because all instructions may throw an
024 * java.lang.VirtualMachineError. These exceptions are omitted.
025 * 
026 * The Lava Language Specification specifies exactly which
027 * <i>RUN-TIME</i> and which <i>LINKING</i> exceptions each
028 * instruction may throw which is reflected by the implementers.  Due
029 * to the structure of the JVM specification, it may be possible that
030 * an Instruction implementing this interface returns a Class[] of
031 * size 0.
032 *
033 * Please note that we speak of an "exception" here when we mean any
034 * "Throwable" object; so this term is equally used for "Exception"
035 * and "Error" objects.
036 *
037 * @version $Id: ExceptionThrower.java 1747278 2016-06-07 17:28:43Z britter $
038 */
039public interface ExceptionThrower {
040
041    java.lang.Class<?>[] getExceptions();
042}
