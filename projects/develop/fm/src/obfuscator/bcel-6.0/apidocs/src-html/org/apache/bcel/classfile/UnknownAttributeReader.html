018package org.apache.bcel.classfile;
019
020/**
021 * Unknown (non-standard) attributes may be read via user-defined factory
022 * objects that can be registered with the Attribute.addAttributeReader
023 * method. These factory objects should implement this interface.
024 * 
025 * @see Attribute
026 * @version $Id: UnknownAttributeReader.java 1747278 2016-06-07 17:28:43Z britter $
027 * @since 6.0
028 */
029public interface UnknownAttributeReader {
030
031    /**
032     * When this attribute reader is added via the static method Attribute.addAttributeReader,
033     * an attribute name is associated with it. As the class file parser parses attributes,
034     * it will call various AttributeReaders based on the name of the attributes it is constructing.
035     * 
036     * @param name_index    An index into the constant pool, indexing a ConstantUtf8
037     *                      that represents the name of the attribute.
038     * @param length        The length of the data contained in the attribute. This is written
039     *                      into the constant pool and should agree with what the factory expects the length to be.
040     * @param file          This is the data input that the factory needs to read its data from.
041     * @param constant_pool This is the constant pool associated with the Attribute that we are constructing.
042     *                      
043     * @return The user-defined AttributeReader should take this data and use
044     * it to construct an attribute.  In the case of errors, a null can be
045     * returned which will cause the parsing of the class file to fail.
046     * 
047     * @see Attribute#addAttributeReader(String, UnknownAttributeReader)
048     */
049    Attribute createAttribute( int name_index, int length, java.io.DataInput file, ConstantPool constant_pool );
050}
